NullTypes

var test = Null # NullTypes are values that are generally used as a placeholder. # When using a NullType as a function argument type, return # type, or variable value, any other type can be placed in without # throwing an error. test = 10 # This will work because test is Null. test = 'hello' # This will not work because test is a string.
